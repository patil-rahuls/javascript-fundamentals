## Nullish Coalescing Assignment Operator `??=`

_This operator not just evaluates, but also, assigns the value._

`??=` prevents evaluating valid values like `0`, `false`, or `""`, which often matter in user input, configs, or database values.

> Example 1: Default configuration values

```javascript
function createUserConfig(config) {
  // config.theme = config.theme ?? "light";
  config.theme ??= "light";
  config.language ??= "en";
  config.showSidebar ??= true;
  return config;
}

console.log(createUserConfig({ language: "fr" }));
/*
{
  language: 'fr',
  theme: 'light',
  showSidebar: true
}
*/
```

&nbsp;

> Example 2: Initializing object properties safely

```javascript
const user = { name: "Rahul" };

// Ensure preferences object always exists
user.preferences ??= {};
user.preferences.darkMode ??= false;

console.log(user);
/*
{
  name: 'Rahul',
  preferences: {
    darkMode: false
  }
}
*/
```

&nbsp;

> Example 3: Caching / Lazy initialization

```javascript
let cache = null;

function getData() {
  cache ??= expensiveDatabaseCall();
  return cache;
}

function expensiveDatabaseCall() {
  console.log("Fetching from DB...");
  return { items: [1, 2, 3] };
}

console.log(getData());
// DB call runs

console.log(getData());
// Cached result, no DB call
```

&nbsp;

> Example 4: Form handling

```javascript
let formData = {
  username: "rahul",
  email: null,
};

formData.email ??= "not-provided@example.com";

console.log(formData);
/*
{
  username: 'rahul',
  email: 'not-provided@example.com'
}
*/
```
